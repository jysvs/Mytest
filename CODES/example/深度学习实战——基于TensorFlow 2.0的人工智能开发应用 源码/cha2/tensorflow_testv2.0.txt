def matrixCompute():
    a = tf.constant([[1,2,4],[3,4,5]])
    b = tf.constant([[2,3],[3,6],[8,9]])
    c = tf.matmul(a,b)
    tfType = type(c)
    tfShape = c.shape
    return c, tfType, tfShape
